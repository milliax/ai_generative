```python
"""Shared Pydantic schemas. ALL inter-agent messages MUST be validated through these."""
from __future__ import annotations

from datetime import date, datetime
from typing import Any, Literal

from pydantic import BaseModel, Field

Urgency = Literal["normal", "rush", "emergency"]


class OrderRequest(BaseModel):
    """客戶原始進單 — Orchestrator 的輸入。"""

    customer: str
    raw_text: str
    received_at: datetime
    urgency: Urgency | None = None


class OrderSpec(BaseModel):
    """Order Intake agent 解析後的結構化規格。"""

    cpu_sku: str
    memory_gb: int = Field(gt=0)
    storage_tb: int = Field(ge=0)
    chassis: str
    quantity: int = Field(gt=0)
    requested_delivery: date
    spec_diff: dict[str, Any] = Field(default_factory=dict)
    urgency: Urgency


class AgentMessage(BaseModel):
    """Agent 之間溝通的訊息。`to_agent=None` 表示送回 orchestrator。"""

    from_agent: str
    to_agent: str | None = None
    payload: dict[str, Any]
    reasoning: str


class CoordinationPlan(BaseModel):
    """系統最終輸出給業務的協調報告。"""

    estimated_price: float
    price_confidence: tuple[float, float]
    estimated_delivery: date
    carbon_footprint_kg: float
    capacity_status: str
    alternative_suppliers: list[dict[str, Any]]
    reference_orders: list[dict[str, Any]]
    risks: list[str]
    next_actions: list[str]
```

```python
"""統一 LLM 呼叫介面。所有 agent 一律經過 call_llm()，不要直接 import litellm。"""
from __future__ import annotations

import logging
import os
import time

from dotenv import load_dotenv
from litellm import completion

load_dotenv()

logger = logging.getLogger(__name__)


class LLMError(RuntimeError):
    """LLM 呼叫失敗（已用盡 retry）。"""


def call_llm(
    messages: list[dict],
    model: str | None = None,
    *,
    temperature: float = 0.2,
    max_retries: int = 3,
    **kwargs,
) -> str:
    """呼叫 LLM，回傳純文字。

    Args:
        messages: OpenAI-style 訊息列表 `[{"role": "user", "content": "..."}]`
        model: 覆寫環境變數 `LLM_MODEL`
        temperature: 預設 0.2（agent 任務需要穩定輸出）
        max_retries: 最多重試次數（指數退避）

    Raises:
        LLMError: 用盡 retry 仍失敗
    """
    model = model or os.getenv("LLM_MODEL", "gemini/gemini-1.5-flash")
    last_err: Exception | None = None

    for attempt in range(max_retries):
        try:
            resp = completion(
                model=model,
                messages=messages,
                temperature=temperature,
                **kwargs,
            )
            content = resp.choices[0].message.content
            logger.info("LLM call ok model=%s tokens≈%d", model, len(content))
            return content
        except Exception as e:
            last_err = e
            wait = 2**attempt
            logger.warning("LLM call failed (attempt %d): %s; retrying in %ds", attempt + 1, e, wait)
            time.sleep(wait)

    raise LLMError(f"LLM call failed after {max_retries} attempts: {last_err}")
```

```python
"""所有 specialist agent 的共同父類。"""
from __future__ import annotations

from abc import ABC, abstractmethod

from shared.llm_client import call_llm
from shared.models import AgentMessage


class BaseAgent(ABC):
    """所有 agent 的基底類別。子類必須：
    - 設定 class attribute `name`（用於 AgentMessage.from_agent）
    - 設定 class attribute `system_prompt`
    - 實作 `run(payload)` 回傳 `AgentMessage`
    """

    name: str = ""
    system_prompt: str = ""

    def call_llm_text(self, user_prompt: str, **kwargs) -> str:
        """便利方法：自動帶上 system prompt。"""
        messages = [
            {"role": "system", "content": self.system_prompt},
            {"role": "user", "content": user_prompt},
        ]
        return call_llm(messages, **kwargs)

    @abstractmethod
    def run(self, payload: dict) -> AgentMessage:
        """執行 agent 的主邏輯。"""
        raise NotImplementedError
```

```python
from __future__ import annotations

from agents.base import BaseAgent
from shared.models import AgentMessage


class CapacityAgent(BaseAgent):
    name = "capacity"
    system_prompt = "You assess factory capacity based on IoT signals."

    HOURS_PER_UNIT_BY_CHASSIS = {"1U": 0.8, "2U": 1.2, "4U": 2.5}

    def run(self, payload: dict) -> AgentMessage:
        qty = int(payload["quantity"])
        chassis = payload.get("chassis", "2U")
        rate = self.HOURS_PER_UNIT_BY_CHASSIS.get(chassis, 1.5)
        hours = qty * rate
        available = hours <= 5000  # 假設單線一週 5000 小時
        return AgentMessage(
            from_agent=self.name,
            to_agent=None,
            payload={
                "available": available,
                "estimated_hours": hours,
                "loading_pct": min(100, hours / 5000 * 100),
            },
            reasoning=f"[W1 stub] {qty} {chassis} × {rate}h = {hours}h; threshold 5000h",
```

```python
"""LangGraph orchestrator — 串接 4 個 specialist agents + aggregator → CoordinationPlan。"""
from __future__ import annotations

from datetime import date
from typing import TypedDict

from langgraph.graph import END, START, StateGraph

from agents.capacity import CapacityAgent
from agents.esg import ESGAgent
from agents.order_intake import OrderIntakeAgent
from agents.pricing import PricingAgent
from shared.models import AgentMessage, CoordinationPlan, OrderRequest


class OrchestratorState(TypedDict):
    request: dict
    spec: dict
    capacity_result: dict
    pricing_result: dict
    esg_result: dict
    trace: list[AgentMessage]


def _intake_node(state: OrchestratorState) -> OrchestratorState:
    msg = OrderIntakeAgent().run({"request": state["request"]})
    state["spec"] = msg.payload["spec"]
    state["trace"].append(msg)
    return state


def _capacity_node(state: OrchestratorState) -> OrchestratorState:
    msg = CapacityAgent().run({
        "quantity": state["spec"]["quantity"],
        "chassis": state["spec"]["chassis"],
    })
    state["capacity_result"] = msg.payload
    state["trace"].append(msg)
    return state


def _pricing_node(state: OrchestratorState) -> OrchestratorState:
    msg = PricingAgent().run({"spec": state["spec"]})
    state["pricing_result"] = msg.payload
    state["trace"].append(msg)
    return state


def _esg_node(state: OrchestratorState) -> OrchestratorState:
    msg = ESGAgent().run({
        "quantity": state["spec"]["quantity"],
        "chassis": state["spec"]["chassis"],
    })
    state["esg_result"] = msg.payload
    state["trace"].append(msg)
    return state


def _aggregate(state: OrchestratorState) -> CoordinationPlan:
    p = state["pricing_result"]
    c = state["capacity_result"]
    e = state["esg_result"]
    risks: list[str] = []
    if not c["available"]:
        risks.append("產能不足，建議啟動替代供應商")
    return CoordinationPlan(
        estimated_price=p["estimated_price"],
        price_confidence=tuple(p["price_confidence"]),
        estimated_delivery=date.fromisoformat(p["estimated_delivery"]),
        carbon_footprint_kg=e["carbon_footprint_kg"],
        capacity_status="OK" if c["available"] else "OVERLOAD",
        alternative_suppliers=e["alternative_suppliers"],
        reference_orders=p["reference_orders"],
        risks=risks,
        next_actions=["業務向客戶確認規格", "確認 BOM 與交期"],
    )


def _build_graph():
    g = StateGraph(OrchestratorState)
    g.add_node("intake", _intake_node)
    g.add_node("capacity", _capacity_node)
    g.add_node("pricing", _pricing_node)
    g.add_node("esg", _esg_node)
    g.add_edge(START, "intake")
    g.add_edge("intake", "capacity")
    g.add_edge("capacity", "pricing")
    g.add_edge("pricing", "esg")
    g.add_edge("esg", END)
    return g.compile()


_GRAPH = _build_graph()


def run_orchestrator(request: OrderRequest) -> tuple[CoordinationPlan, list[AgentMessage]]:
    """執行完整 pipeline，回傳協調計畫 + agent trace。"""
    initial: OrchestratorState = {
        "request": request.model_dump(mode="json"),
        "spec": {},
        "capacity_result": {},
        "pricing_result": {},
        "esg_result": {},
        "trace": [],
    }
    final = _GRAPH.invoke(initial)
    plan = _aggregate(final)
    return plan, final["trace"]
```

```python
from datetime import date, timedelta
from pathlib import Path

import pandas as pd

CPU_SKUS = ["Xeon-8480", "Xeon-9654", "EPYC-9554", "EPYC-9654", "Xeon-6442Y"]

    price = memory * qty * rng.uniform(80, 120)

        "order_id": f"HIST-{i:04d}",
        "customer": customer,
        "cpu_sku": cpu,
        "memory_gb": memory,
        "storage_tb": storage,
        "chassis": chassis,
        "quantity": qty,
        "delivered_at": delivered.isoformat(),
        "final_price": round(price, 2),
        "carbon_kg": round(carbon, 2),
        "spec_summary": f"{customer} {qty}× {chassis} {cpu} {memory}GB/{storage}TB",
```

```python
"""把歷史訂單 CSV 灌進 ChromaDB（spec_summary 做 embedding，其他欄位放 metadata）。"""
from __future__ import annotations

import os
from pathlib import Path

import chromadb
import pandas as pd
from chromadb.utils.embedding_functions import SentenceTransformerEmbeddingFunction

COLLECTION = "historical_orders"
_EMBED = SentenceTransformerEmbeddingFunction(model_name="all-MiniLM-L6-v2")


def ingest_orders_to_chroma(csv_path: Path | str, persist_dir: Path | str | None = None) -> int:
    persist_dir = str(persist_dir or os.getenv("CHROMA_PERSIST_DIR", "./chroma_db"))
    client = chromadb.PersistentClient(path=persist_dir)
    try:
        client.delete_collection(COLLECTION)
    except Exception:
        pass
    col = client.create_collection(COLLECTION, embedding_function=_EMBED)

    df = pd.read_csv(csv_path)
    col.add(
        ids=df["order_id"].astype(str).tolist(),
        documents=df["spec_summary"].astype(str).tolist(),
        metadatas=df.drop(columns=["spec_summary"]).to_dict(orient="records"),
    )
    return len(df)
```

```python
"""依規格摘要檢索 top-K 相似歷史訂單。"""
from __future__ import annotations

import os
from pathlib import Path

import chromadb
from chromadb.utils.embedding_functions import SentenceTransformerEmbeddingFunction

from pricing.ingest import COLLECTION

_EMBED = SentenceTransformerEmbeddingFunction(model_name="all-MiniLM-L6-v2")


def retrieve_similar(
    query: str, k: int = 5, persist_dir: Path | str | None = None
) -> list[dict]:
    persist_dir = str(persist_dir or os.getenv("CHROMA_PERSIST_DIR", "./chroma_db"))
    client = chromadb.PersistentClient(path=persist_dir)
    col = client.get_collection(COLLECTION, embedding_function=_EMBED)
    res = col.query(query_texts=[query], n_results=k)

    hits: list[dict] = []
    for i, meta in enumerate(res["metadatas"][0]):
        dist = res["distances"][0][i]
        hits.append({
            **meta,
            "similarity": max(0.0, 1.0 - dist),  # cosine distance → similarity
        })
    return hits
```
